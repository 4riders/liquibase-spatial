50     @Override
51     public Sql[] generateSql(final CreateTableStatement statement, final Database database,
52           final SqlGeneratorChain sqlGeneratorChain) {
53        final List<Sql> list = new ArrayList<Sql>(Arrays.asList(sqlGeneratorChain.generateSql(
54              statement, database)));
55        for (final Entry<String, LiquibaseDataType> entry : statement.getColumnTypes().entrySet()) {
56           if (entry.getValue() instanceof GeometryType) {
57              final String schemaName = statement.getSchemaName();
58              final String escapedSchemaName = schemaName == null ? "NULL" : "'"
59                    + database.escapeStringForDatabase(schemaName) + "'";
60              final String tableName = statement.getTableName();
61              final String columnName = entry.getKey();
62  
63              final GeometryType geometryType = (GeometryType) entry.getValue();
64              final Integer srid = geometryType.getSRID() == null ? 4326 : geometryType.getSRID();
65              final String geomType = geometryType.getGeometryType() == null ? "'Geometry'" : "'"
66                    + database.escapeStringForDatabase(geometryType.getGeometryType()) + "'";
67  
68              final String sql = "CALL AddGeometryColumn(" + escapedSchemaName + ", '" + tableName
69                    + "', '" + columnName + "', " + srid + ", " + geomType + ", 2)";
70              final Sql addGeometryColumn = new UnparsedSql(sql);
71              list.add(addGeometryColumn);
72           }
73        }
74        return list.toArray(new Sql[list.size()]);
75     }
