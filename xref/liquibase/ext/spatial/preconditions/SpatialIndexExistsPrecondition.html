113    @Override
114    public void check(final Database database, final DatabaseChangeLog changeLog,
115          final ChangeSet changeSet) throws PreconditionFailedException, PreconditionErrorException {
116       Precondition delegatedPrecondition;
117       if (database instanceof DerbyDatabase || database instanceof H2Database) {
118          final TableExistsPrecondition precondition = new TableExistsPrecondition();
119          precondition.setCatalogName(getCatalogName());
120          precondition.setSchemaName(getSchemaName());
121          final String tableName = getHatboxTableName();
122          precondition.setTableName(tableName);
123          delegatedPrecondition = precondition;
124       } else {
125          final IndexExistsPrecondition precondition = new IndexExistsPrecondition();
126          precondition.setCatalogName(getCatalogName());
127          precondition.setSchemaName(getSchemaName());
128          precondition.setTableName(getTableName());
129          precondition.setIndexName(getIndexName());
130          precondition.setColumnNames(getColumnNames());
131          delegatedPrecondition = precondition;
132       }
133       delegatedPrecondition.check(database, changeLog, changeSet);
134    }

142       final String tableName;

152     * Creates an example of the database object for which to check.

154     * @param database
155     *           the database instance.
156     * @param tableName
157     *           the table name of the index.
158     * @return the database object example.
159     */
160    public DatabaseObject getExample(final Database database, final String tableName) {
161       final Schema schema = new Schema(getCatalogName(), getSchemaName());
162       final DatabaseObject example;
163 
164       // For GeoDB, the index is another table.
165       if (database instanceof DerbyDatabase || database instanceof H2Database) {
166          final String correctedTableName = database.correctObjectName(getHatboxTableName(),
167                Table.class);
168          example = new Table().setName(correctedTableName).setSchema(schema);
169       } else {
170          example = getIndexExample(database, schema, tableName);
171       }
172       return example;
173    }

175    /**
176     * Generates the {@link Index} example (taken from {@link IndexExistsPrecondition}).
177     * 
178     * @param database
179     *           the database instance.
180     * @param schema
181     *           the schema instance.
182     * @param tableName
183     *           the table name of the index.
184     * @return the index example.
185     */
186    protected Index getIndexExample(final Database database, final Schema schema,
187          final String tableName) {
188       final Index example = new Index();
189       if (tableName != null) {
190          example.setTable((Table) new Table().setName(
191                database.correctObjectName(getTableName(), Table.class)).setSchema(schema));
192       }
193       example.setName(database.correctObjectName(getIndexName(), Index.class));
194       if (StringUtils.trimToNull(getColumnNames()) != null) {
195          for (final String column : getColumnNames().split("\\s*,\\s*")) {
196             example.getColumns().add(database.correctObjectName(column, Column.class));
197          }
198       }
199       return example;
200    }
