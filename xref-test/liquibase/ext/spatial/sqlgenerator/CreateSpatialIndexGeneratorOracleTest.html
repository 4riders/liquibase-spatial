66     /**
67      * Tests
68      * {@link CreateSpatialIndexGeneratorOracle#generateSql(CreateSpatialIndexStatement, Database, SqlGeneratorChain)}
69      * with a variety of inputs.
70      * 
71      * @param statement
72      */
73     @Test(dataProvider = "generateSqlTestData")
74     public void testGenerateSql(final CreateSpatialIndexStatement statement) {
75        final CreateSpatialIndexGeneratorOracle generator = new CreateSpatialIndexGeneratorOracle();
76        final Database database = new OracleDatabase();
77        final SqlGeneratorChain sqlGeneratorChain = mock(SqlGeneratorChain.class);
78        final Sql[] result = generator.generateSql(statement, database, sqlGeneratorChain);
79        assertNotNull(result);
80        assertEquals(result.length, 3);
81  
82        // Verify the DELETE statement.
83        final String deleteSql = result[0].toSql();
84        String deletePattern = "(?i)DELETE FROM user_sdo_geom_metadata ";
85        deletePattern += "WHERE table_name = '" + statement.getTableName().toUpperCase();
86        deletePattern += "' AND column_name = '" + statement.getColumns()[0].toUpperCase();
87        deletePattern += "'";
88        assertTrue(deleteSql.matches(deletePattern), "'" + deleteSql
89              + "' does not match the pattern '" + deletePattern + "'");
90        assertNotNull(result[0].getAffectedDatabaseObjects());
91        assertTrue(result[0].getAffectedDatabaseObjects().size() >= 1, result[0]
92              .getAffectedDatabaseObjects().toString());
93  
94        // Verify the INSERT statement.
95        final String insertSql = result[1].toSql();
96        final String insertPattern = "(?i)INSERT INTO user_sdo_geom_metadata \\(.*\\) VALUES \\(.*\\)";
97        assertTrue(insertSql.matches(insertPattern), "'" + insertSql
98              + "' does not match the pattern '" + insertPattern + "'");
99        assertNotNull(result[1].getAffectedDatabaseObjects());
100       assertTrue(result[1].getAffectedDatabaseObjects().size() >= 1, result[0]
101             .getAffectedDatabaseObjects().toString());
102 
103       // Verify the CREATE INDEX statement.
104       final String createSql = result[2].toSql();
105       String createPattern = "(?i)CREATE INDEX ";
106       if (statement.getTableCatalogName() != null) {
107          createPattern += statement.getTableCatalogName() + '.';
108       } else if (statement.getTableSchemaName() != null) {
109          createPattern += statement.getTableSchemaName() + '.';
110       }
111       createPattern += statement.getIndexName() + " ON ";
112       if (statement.getTableCatalogName() != null) {
113          createPattern += statement.getTableCatalogName() + '.';
114       } else if (statement.getTableSchemaName() != null) {
115          createPattern += statement.getTableSchemaName() + '.';
116       }
117       createPattern += statement.getTableName() + " \\(" + statement.getColumns()[0] + "\\)";
118       createPattern += " INDEXTYPE IS mdsys.spatial_index";
119       if (statement.getGeometryType() != null || statement.getTablespace() != null) {
120          createPattern += " PARAMETERS \\('";
121          if (statement.getGeometryType() != null) {
122             createPattern += " ?layer_gtype=[a-zA-Z]+";
123          }
124          if (statement.getTablespace() != null) {
125             createPattern += " ?tablespace=" + statement.getTablespace();
126          }
127          createPattern += "'\\)";
128       }
129       assertTrue(createSql.matches(createPattern), "'" + createSql
130             + "' does not match the pattern '" + createPattern + "'");
131       assertNotNull(result[2].getAffectedDatabaseObjects());
132       assertTrue(result[2].getAffectedDatabaseObjects().size() > 1, result[0]
133             .getAffectedDatabaseObjects().toString());
134    }
