23  /**
24   * <code>AddSpatialColumnGeneratorGeoDB</code> ... <code>DropSpatialColumnGeneratorGeoDB</code> is a
25   * {@link DropColumnGenerator} that specializes in GeoDB. If the column to be dropped is the
26   * geometry column, the <code>DropGeometryColumn</code> is invoked instead of performing the typical
27   * <code>ALTER TABLE</code> statement. Otherwise, the next SQL generator in the chain is invoked to
28   * handle the request.
29   */
30  public class AddSpatialColumnGeneratorGeoDB extends AbstractSqlGenerator<AddColumnStatement> {
31     /**
32      * @see liquibase.sqlgenerator.core.AbstractSqlGenerator#supports(liquibase.statement.SqlStatement,
33      *      liquibase.database.Database)
34      */
35     @Override
36     public boolean supports(final AddColumnStatement statement, final Database database) {
37        return database instanceof DerbyDatabase || database instanceof H2Database;
38     }
39  
40     /**
41      * @see liquibase.sqlgenerator.core.AbstractSqlGenerator#getPriority()
42      */
43     @Override
44     public int getPriority() {
45        return SqlGenerator.PRIORITY_DATABASE + 1;
46     }

48     @Override
49     public ValidationErrors validate(final AddColumnStatement statement, final Database database,
50           final SqlGeneratorChain sqlGeneratorChain) {
51        final ValidationErrors errors = new ValidationErrors();
52        final LiquibaseDataType dataType = DataTypeFactory.getInstance().fromDescription(
53              statement.getColumnType());
54  
55        // Ensure that the SRID parameter is provided.
56        if (dataType instanceof GeometryType) {
57           final GeometryType geometryType = (GeometryType) dataType;
58           if (geometryType.getSRID() == null) {
59              errors.addError("The SRID parameter is required on the geometry type");
60           }
61        }
62        final ValidationErrors chainErrors = sqlGeneratorChain.validate(statement, database);
63        if (chainErrors != null) {
64           errors.addAll(chainErrors);
65        }
66        return errors;
67     }

69     @Override
70     public Sql[] generateSql(final AddColumnStatement statement, final Database database,
71           final SqlGeneratorChain sqlGeneratorChain) {
72  
73        GeometryType geometryType = null;
74        final LiquibaseDataType dataType = DataTypeFactory.getInstance().fromDescription(
75              statement.getColumnType());
76        if (dataType instanceof GeometryType) {
77           geometryType = (GeometryType) dataType;
78        }
79  
80        final boolean isGeometryColumn = geometryType != null;
81  
82        // The AddGeometryColumn procedure handles the column already being present, so let a
83        // downstream SQL generator handle the typical column addition logic (e.g. placement in the
84        // table) then invoke the procedure.
85        final List<Sql> list = new ArrayList<Sql>();
86        list.addAll(Arrays.asList(sqlGeneratorChain.generateSql(statement, database)));
87        if (isGeometryColumn) {
88           String schemaName = statement.getSchemaName();
89           if (schemaName == null) {
90              schemaName = database.getDefaultSchemaName();
91           }
92           final String tableName = statement.getTableName();
93           final String columnName = statement.getColumnName();
94  
95           final int srid = geometryType.getSRID();
96           final String geomType = StringUtils.trimToNull(geometryType.getGeometryType()) == null ? "'Geometry'"
97                 : "'" + database.escapeStringForDatabase(geometryType.getGeometryType()) + "'";
98           final String sql = "CALL AddGeometryColumn('" + schemaName + "', '" + tableName + "', '"
99                 + columnName + "', " + srid + ", " + geomType + ", 2)";
100          final Sql addGeometryColumn = new UnparsedSql(sql);
101          list.add(addGeometryColumn);
102       }
103       return list.toArray(new Sql[list.size()]);
104    }
